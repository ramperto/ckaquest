Fix the manifest directly:

  sudo tee /etc/kubernetes/manifests/infra-monitor.yaml > /dev/null <<'EOF'
  apiVersion: v1
  kind: Pod
  metadata:
    name: infra-monitor
    namespace: kube-system
  spec:
    restartPolicy: Always
    hostNetwork: true
    containers:
      - name: monitor
        image: busybox:1.36
        command: ["sh", "-c", "sleep 3600"]
        resources:
          requests:
            cpu: 10m
            memory: 16Mi
  EOF

  # Watch for the pod to appear (with node name suffix)
  NODE=$(kubectl get nodes -o jsonpath='{.items[0].metadata.name}')
  kubectl get pod infra-monitor-$NODE -n kube-system -w
