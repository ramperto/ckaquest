Fix the pod to stay within the LimitRange bounds:

  kubectl delete pod highcpu -n ckaquest

  cat <<EOF | kubectl apply -f -
  apiVersion: v1
  kind: Pod
  metadata:
    name: highcpu
    namespace: ckaquest
    labels:
      app: highcpu
  spec:
    containers:
      - name: app
        image: nginx:1.25
        resources:
          requests:
            cpu: "100m"
            memory: "64Mi"
          limits:
            cpu: "200m"
            memory: "128Mi"
  EOF
