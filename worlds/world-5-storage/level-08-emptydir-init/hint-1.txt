Check the pod status:

  kubectl get pod worker -n ckaquest
  # STATUS: CrashLoopBackOff or Error

  kubectl logs worker -n ckaquest -c main
  # cat: can't open '/data/config.json': No such file or directory

Check both volume mount paths:
  kubectl get pod worker -n ckaquest \
    -o jsonpath='{range .spec.initContainers[*]}{.name}: {.volumeMounts[*].mountPath}{"\n"}{end}'
  # init-config: /work

  kubectl get pod worker -n ckaquest \
    -o jsonpath='{range .spec.containers[*]}{.name}: {.volumeMounts[*].mountPath}{"\n"}{end}'
  # main: /data   ← mismatch!
