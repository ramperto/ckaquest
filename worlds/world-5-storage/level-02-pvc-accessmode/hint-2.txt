PVC spec is immutable — delete and recreate:

  kubectl delete pvc pipeline-data -n ckaquest

  kubectl apply -f - <<EOF
  apiVersion: v1
  kind: PersistentVolumeClaim
  metadata:
    name: pipeline-data
    namespace: ckaquest
  spec:
    accessModes:
      - ReadWriteOnce   # must match PV's accessModes
    storageClassName: manual
    resources:
      requests:
        storage: 5Gi
  EOF
