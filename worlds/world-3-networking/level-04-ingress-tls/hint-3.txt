Complete solution:

  # 1. Generate self-signed cert
  openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
    -keyout /tmp/webapp.key \
    -out /tmp/webapp.crt \
    -subj "/CN=webapp.local/O=CKAQuest"

  # 2. Create the TLS secret
  kubectl create secret tls webapp-tls \
    --cert=/tmp/webapp.crt \
    --key=/tmp/webapp.key \
    -n ckaquest

  # 3. Verify
  kubectl get secret webapp-tls -n ckaquest -o yaml | head -20
