Or create directly with YAML:

  cat <<EOF | kubectl apply -f -
  apiVersion: v1
  kind: Service
  metadata:
    name: webapp-svc
    namespace: ckaquest
  spec:
    type: NodePort
    selector:
      app: webapp
    ports:
      - port: 80
        targetPort: 80
        nodePort: 30080
  EOF

Test it:
  NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[0].address}')
  curl http://$NODE_IP:30080
