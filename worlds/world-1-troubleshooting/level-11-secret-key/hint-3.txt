Delete the broken pod and recreate with the correct key:

  kubectl delete pod app -n ckaquest

  cat <<EOF | kubectl apply -f -
  apiVersion: v1
  kind: Pod
  metadata:
    name: app
    namespace: ckaquest
    labels:
      app: myapp
  spec:
    containers:
      - name: app
        image: nginx:1.25
        env:
          - name: DB_PASSWORD
            valueFrom:
              secretKeyRef:
                name: db-creds
                key: password
  EOF
