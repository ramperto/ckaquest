Delete and recreate the pod with a reasonable ephemeral storage request:

  kubectl delete pod storage-hog -n ckaquest

  kubectl apply -f - <<EOF
  apiVersion: v1
  kind: Pod
  metadata:
    name: storage-hog
    namespace: ckaquest
    labels:
      app: storage-hog
  spec:
    containers:
      - name: busybox
        image: busybox:1.36
        command: ["sleep", "3600"]
        resources:
          requests:
            ephemeral-storage: 500Mi
  EOF
