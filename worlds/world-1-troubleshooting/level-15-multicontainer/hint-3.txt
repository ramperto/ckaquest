Delete and recreate with a fixed sidecar command:

  kubectl delete pod web-with-sidecar -n ckaquest

  cat <<EOF | kubectl apply -f -
  apiVersion: v1
  kind: Pod
  metadata:
    name: web-with-sidecar
    namespace: ckaquest
    labels:
      app: web
  spec:
    volumes:
      - name: shared-logs
        emptyDir: {}
    containers:
      - name: web
        image: nginx:1.25
        volumeMounts:
          - name: shared-logs
            mountPath: /var/log/nginx
      - name: log-collector
        image: busybox:1.36
        command: ["sh", "-c", "while [ ! -f /var/log/nginx/access.log ]; do sleep 2; done; tail -f /var/log/nginx/access.log"]
        volumeMounts:
          - name: shared-logs
            mountPath: /var/log/nginx
  EOF
