Pods are immutable for spec.affinity — delete and recreate:

  kubectl delete pod zone-aware-app -n ckaquest

  kubectl apply -f - <<EOF
  apiVersion: v1
  kind: Pod
  metadata:
    name: zone-aware-app
    namespace: ckaquest
    labels:
      app: zone-aware-app
  spec:
    affinity:
      nodeAffinity:
        requiredDuringSchedulingIgnoredDuringExecution:
          nodeSelectorTerms:
            - matchExpressions:
                - key: zone
                  operator: In
                  values:
                    - us-west   # fixed: matches actual node label
    containers:
      - name: app
        image: busybox:1.36
        command: ["sh", "-c", "echo 'Running in correct zone'; sleep 3600"]
  EOF
