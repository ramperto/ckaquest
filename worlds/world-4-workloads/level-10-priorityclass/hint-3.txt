Full fix:

  # 1. Create the PriorityClass
  kubectl apply -f - <<EOF
  apiVersion: scheduling.k8s.io/v1
  kind: PriorityClass
  metadata:
    name: high-priority
  value: 1000000
  globalDefault: false
  EOF

  # 2. Delete and recreate the pod so the scheduler re-evaluates it
  kubectl delete pod critical-app -n ckaquest

  kubectl apply -f - <<EOF
  apiVersion: v1
  kind: Pod
  metadata:
    name: critical-app
    namespace: ckaquest
    labels:
      app: critical-app
  spec:
    priorityClassName: high-priority
    containers:
      - name: app
        image: nginx:1.25
        ports:
          - containerPort: 80
  EOF

  # 3. Verify
  kubectl get pod critical-app -n ckaquest
  kubectl get priorityclass high-priority
